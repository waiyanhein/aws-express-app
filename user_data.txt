#!/bin/bash

curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.38.0/install.sh | bash

source ~/.bashrc

nvm install 16

nvm use 16

sudo yum update -y

sudo yum install git -y

cd /home/ec2-user

git clone https://github.com/waiyanhein/express-app.git

cd express-app

sudo chmod -R 755 .

npm i

node app.js > app.out.log 2> app.err.log < /dev/null &
